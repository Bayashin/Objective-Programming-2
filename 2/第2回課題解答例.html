#!/usr/bin/env python
# -*- coding: utf-8 -*-
import csv

def lecture02_01_printHeroStatus() -> None:

    # ヒーローデータの読み込み
    hero_header = []
    hero_data = []
    with open('lecture02_Hero.csv') as f:
        reader = csv.reader(f)
        for row in reader:
            if len(hero_header) == 0:
                hero_header = row
            else:
                hero_data.append(row)

    # 出力処理
    for hero in hero_data:
        if hero[0] == '1': # idが1の場合
            print(f"{hero[1]}のステータスは" +
            f"HP:{hero[2]}," +
            f"MP:{hero[3]}," +
            f"Atk:{hero[4]}," +
            f"Def:{hero[5]}," +
            f"Age:{hero[6]}")

#!/usr/bin/env python
# -*- coding: utf-8 -*-
import xml.etree.ElementTree as ET

def lecture02_02():
    root = ET.Element('book')
    # ET.SubElementで子要素を作成
    article = ET.SubElement(root, 'article')
    article.attrib['title']='卒業論文'
    # idについては１から昇順にならんでいるので先に作成しておく
    for i in range(6):
        chapter = ET.SubElement(article, 'chapter')
        chapter.attrib['id'] = str(i+1)

    # 木構造の子要素へのアクセスには，リストにインデックスでアクセスするように書く
    article[0].attrib['name'] = 'はじめに'
    article[0].attrib['pages'] = str(2)
    article[1].attrib['name'] = '基礎理論'
    article[1].attrib['pages'] = str(8)
    article[2].attrib['name'] = '実験方法'
    article[2].attrib['pages'] = str(6)
    article[3].attrib['name'] = '結果と考察'
    article[3].attrib['pages'] = str(2)
    article[4].attrib['name'] = 'まとめ'
    article[4].attrib['pages'] = str(1)
    article[5].attrib['name'] = '参考文献'
    article[5].attrib['pages'] = str(2)

    # novelの子要素を作成する．木の深さはarticleと同じでbookの直下
    novel = ET.SubElement(root, 'novel')
    # ここでは講義資料に基づいて子要素をリストに追加することにする
    novel_chapters = []
    for i in range(5):
        chapter = ET.SubElement(novel, 'chapter')
        chapter.attrib['id'] = str(i+1)
        novel_chapters.append(chapter)

    # 子要素が入ったリストから要素をリストのインデックスで呼び出し修正
    novel_chapters[0].attrib['name'] = '1日のはじまり'
    novel_chapters[0].attrib['pages'] = str(2)
    novel_chapters[1].attrib['name'] = '朝食'
    novel_chapters[1].attrib['pages'] = str(8)
    novel_chapters[2].attrib['name'] = '仕事'
    novel_chapters[2].attrib['pages'] = str(6)
    novel_chapters[3].attrib['name'] = '帰宅後'
    novel_chapters[3].attrib['pages'] = str(2)
    novel_chapters[4].attrib['name'] = '新しい朝'
    novel_chapters[4].attrib['pages'] = str(1)

    # write readable xml (have your attention to 'wb')
    with open('lecture02_data.xml', 'wb') as f:
        import xml.dom.minidom as md
        f.write(md.parseString(ET.tostring(root, encoding='utf-8', xml_declaration=True)).toprettyxml(indent='  ',encoding="utf-8"))


if __name__ == '__main__':
    lecture02_02()

評価方法

  1. タイポ，インデントのずれなどプログラムとして実行できないものは評価しない．
  2. この課題については，簡易なミス（例えば，要件を１つ忘れていたなど）に限り減点はしない．
  3. 遅れて提出された課題の採点はせず，提出点のみとする．
