課題1

以下に示すFizzBuzzBaseクラスはFizzBuzzのゲームの仕様を示す基底クラスです．

FizzBuzzBaseクラスを継承してFizzBuzzクラスを作成してください．

import abc

class FizzBuzzBase:

    __DEFAULT_LOOP_TIME = 20

    def __init__(self, max_count=__DEFAULT_LOOP_TIME):
        self.print_fizzbuzz(max_count)

    @abc.abstractmethod
    def __convert(self, input : int) -> str:
        """
        数値をFizzBuzzルールに従って文字列に変換する処理

        Parameters
        ----------
        input : int
            正の整数
        
        Returns
        -------
        result_str : str
            FizzBuzzルールに従って変換した文字列
        """
        raise NotImplementedError()

    @abc.abstractmethod
    def print_fizzbuzz(self, max_count : int) -> None:
        """
        結果を出力する処理
        max_countには1以上の整数が入力されることを前提としているので例外処理は不要

        Parameters
        ----------
        max_count : int
            1からmax_countまで__convert処理を繰り返す
            __convertから得られた文字列をprintする
        """
        raise NotImplementedError()

以下の要件を満たすこと

  1. FizzBuzzクラスのコンストラクタからsuperを使ってFizzBuzzBaseクラスのコンストラクタを呼び出す

  2. FizzBuzzBaseクラス内のコメントを参照してFizzBuzzクラスに抽象メソッドを実装する

ヒント

FizzBuzz実装のヒントにしてください

max_count=10
for i in range(max_count):
    if i%15 == 0:
        print("FizzBuzz")
    else if i%3 == 0:
        print("Fizz")
    else if i%5 == 0:
        print("Buzz")
    else:
        print(i)
def __convert(input : int) -> str:
    result_str = "FizzBuzz" if input%15==0 else "Fizz" if input%3==0 else "Buzz" if input%5==0 else f"{input}"
    return result_str

# 最大値を設定
max_count=10

# print処理
number_list : list[int] = list(range(max_count))
number_list = list(map(lambda x: x+1, number_list))
for x in map(__convert, number_list):
    print(x)

# print処理4行は以下のように2行にまとめてもよい
for x in map(__convert, map(lambda x: x+1, range(max_count))):
    print(x)
for x in [__convert(input) for input in [x+1 for x in range(max_count)]]:
    print(x)
